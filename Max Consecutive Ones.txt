class Solution{
    public int findMaxConsecutiveOnes(int[]nums){
        int cnt = 0;
        int maxi = 0;
        for(int i =0;i<nums.length;i++)
        {
            if(nums[i]==1)
            {
                cnt++;
            }
            else
            {
                maxi = Math.max(cnt,maxi);
            }
        }
        return maxi;
    }
}
